## 自动化测试脚本

### Bash测试脚本示例
```bash
#!/bin/bash

# 中国滑雪积分系统 - 页面可用性测试脚本

BASE_URL="https://gaohao10010111-sketch.github.io/ski"

PAGES=(
  "/"
  "/about"
  "/scoring-systems"
  "/points"
  "/points/calculator"
  "/points/fis"
  "/points/rankings"
  "/points/trends"
  "/athletes"
  "/competitions"
  "/competitions/schedule"
  "/competitions/stats"
  "/registration"
  "/registration/fees"
  "/registration/manage"
  "/registration/online"
  "/rules"
  "/rules/competition"
  "/rules/fis"
  "/rules/points"
  "/results-announcement"
  "/results-import"
  "/results-query"
  "/login"
  "/register"
  "/profile"
  "/test-accounts"
  "/test-simple"
)

echo "开始测试..."
echo "基础URL: $BASE_URL"
echo "========================================"

SUCCESS=0
FAIL=0

for page in "${PAGES[@]}"; do
  url="${BASE_URL}${page}"
  status=$(curl -o /dev/null -s -w "%{http_code}" "$url")

  if [ "$status" -eq 200 ]; then
    echo "✅ $page - 状态码: $status"
    ((SUCCESS++))
  else
    echo "❌ $page - 状态码: $status"
    ((FAIL++))
  fi
done

echo "========================================"
echo "测试完成！"
echo "成功: $SUCCESS 页面"
echo "失败: $FAIL 页面"
echo "总计: $((SUCCESS + FAIL)) 页面"
```

### 使用方法
```bash
# 保存为 test-pages.sh
chmod +x test-pages.sh
./test-pages.sh
```
